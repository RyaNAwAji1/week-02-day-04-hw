package com.example.demo.Controller;

import com.example.demo.Service.CarService;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.Errors;
import org.springframework.web.bind.annotation.*;

import javax.validation.Valid;
@RestController
@RequestMapping("api/v1/car")
@RequiredArgsConstructor
public class Car {
    private final CarService carService;
    @GetMapping("car/")
    public ResponseEntity ArrayList<car> getCar(){
        return ResponseEntity.status(200).body(carService.getUsers());

    }
    @PostMapping("car/1")
    public ResponseEntity<Api> addBook(@RequestBody @Valid User user, Errors errors) {

        if(errors.hasErrors()){
            return ResponseEntity.status(400).body(new Api(errors.getFieldError().getDefaultMessage(),400));
        }
        return ResponseEntity.status(200).body(new Api("New car added",200));

        boolean isUserAdded=carService.addUcar(carService);
        if(!isUserAdded){
            return ResponseEntity.status(500).body(new Api("Error adding a car",500));

        }
        return ResponseEntity.status(200).body(new Api("New car added",200));
    }

    public CarService getCarService() {
        return carService;
    }
}
