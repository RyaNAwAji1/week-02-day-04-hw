package com.example.demo.Controller;

import com.example.demo.Service.OrderService;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.Errors;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;

import javax.validation.Valid;

public class Order {
    
    private final OrderService orderService;


    @PostMapping
    public ResponseEntity<Api> addOrder(@RequestBody @Valid Order order, Errors errors) {
        if (errors.hasErrors()) {
            return ResponseEntity.status(400).body(new Api(errors.getFieldError().getDefaultMessage(), 400));
        }


        @PutMapping("order/")
        public ResponseEntity addOrder (@PathVariable Integer ID,
                @PathVariable String userID,
                @PathVariable String carID)
        {
            if (!orderService.addOrder(carid, userid)) {
                return ResponseEntity.status(400).body("");
            }
            return ResponseEntity.status(201).body(new Api("New Order added", 201));
        }
    }
